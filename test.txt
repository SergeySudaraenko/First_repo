def total_salary(path):
    try:
        with open(path, 'r', encoding='utf-8') as file:
            lines = file.readlines()

        total_salary = 0
        num_developers = 0

        for line in lines:
            name, salary_str = line.strip().split(',')
            salary = int(salary_str)
            total_salary += salary
            num_developers += 1

        if num_developers == 0:
            return 0, 0  # Якщо в файлі немає розробників, повернути (0, 0) для уникнення ділення на нуль.

        average_salary = total_salary / num_developers

        return total_salary, average_salary

    except FileNotFoundError:
        print(f"Файл '{path}' не знайдено.")
        return None

    except Exception as e:
        print(f"Виникла помилка: {e}")
        return None

# Приклад виклику функції:
result = total_salary('шлях_до_вашого_файлу.txt')

if result is not None:
    total, average = result
    print(f"Загальна сума зарплат: {total}")
    print(f"Середня зарплата: {average}")
